function souge(A, B) {
    //在B组里搜含A字
    var C = new Array,
        D = new Array;
    A = escape(A).replace(/%/g, "\\");
    eval("var re=/" + A + "/;");
    for (var i = 0; i < B.length; i++) {
        if (re.test(B[i])) {
            C.push(i);
            D.push(B[i]);
        }
    }
    return [C, D];
}
function saoge(dirs) {
    var Paths = new Array,
        Files = new Array;
    for (var i = 0; i < dirs.length; i++) {
        if (files.isDir(dirs[i])) {
            var File = files.listDir(dirs[i], function(name) {
                return (name.endsWith(".m4a") || name.endsWith(".mp3")) && files.isFile(files.join(dirs[i], name));
            });
            if (File.length > 0) {
                Files = Files.concat(File.join("☆").split("☆"));
                Paths = Paths.concat(shuzuG(File, dirs[i]));
            }
        }
    }
    if (Files.length > 0) {
        return [Paths, Files];
    }
}

function dir_abcd(a, b, c, d) {
    var A = a.split(b);
    var K = new Array;
    var kg = false;
    for (var i = 0; i < A.length; i++) {
        if (i == c || i == A.length + c) {
            kg = true
        };
        if (kg) {
            K.push(A[i])
        };
        if (i == d || i == A.length + d) {
            kg = false
        };
    }
    return b + K.join(b);
