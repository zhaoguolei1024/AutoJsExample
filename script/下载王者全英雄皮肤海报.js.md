console.show();

var dirpath = "/sdcard/王者荣耀英雄海报/";
files.ensureDir(dirpath);

var url = 'http://pvp.qq.com/web201605/js/herolist.json';
var head = {
    'User-Agent': 'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/72.0.3626.109 Safari/537.36'
};
var res = http.get(url, head);

var json = res.body.json();
toastLog("请等待，正在下载");

for (let j in json) {
    let obj = json[j];
    let I = obj.ename;
    let name_ary = obj.skin_name.split("|");
    log("第", Number(j) + 1, json.length, "英雄");
    log("皮肤数量", name_ary.length);
    for (let n in name_ary) {
        let heroskin_links = "http://game.gtimg.cn/images/yxzj/img201606/skin/hero-info/" + I + "/" + I + "-bigskin-" + (Number(n) + 1) + ".jpg";
        let res = http.get(heroskin_links);
        let path = "/" + obj.cname + "_" + name_ary[n] + (Number(n) + 1) + '.jpg';
        if (res.statusCode == 200) {
            files.writeBytes(files.join(dirpath, path), res.body.bytes());
            media.scanFile(files.join(dirpath, path));
            log("OK " + path);
        } else {
            log("NO " + path);
        };
    };
};
toastLog("全部下载完成");
