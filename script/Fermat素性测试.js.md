function 素性测试(input, turn) {
  for (var i = 0; i < (turn || 4); i++) {
    if (幂模(random(1, input - 1), input - 1, input) !== 1) {
      return false
    }
  }
  return true

  function 幂模(C, E, N) {
    var i = C % N,
      D = 1
    while (E > 1) {
      if ((E & 1) !== 0) {
        D = D * i % N
      }
      i = i * i % N
      E >>= 1
    }
    return i * D % N
  }
}```
