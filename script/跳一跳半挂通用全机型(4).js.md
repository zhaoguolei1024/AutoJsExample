        case event.ACTION_DOWN:

            y = event.getRawY();
            windowX = window.getX();
            windowY = window.getY();
            downTime = new Date().getTime();

            //移动手指时调整悬浮窗位置

                windowY + (event.getRawY() - y));
            
            return true;
