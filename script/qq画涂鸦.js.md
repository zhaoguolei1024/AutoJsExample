            sf = 100 / img.width
        }
        img = images.scale(img, sf, sf)
        img = images.grayscale(img)
        img = images.adaptiveThreshold(img, 200, "MEAN_C", "BINARY", 25, 10);
        img = images.adaptiveThreshold(img, 200, "MEAN_C", "BINARY", 3, 3);

        var h = bitmap.getHeight();

function 画图(内容, w, h) {
    var 笔画和 = []
    var 笔画 = []
    //-3618616, -16777216
    fx = [
        [1, 0],
        [0, 1],
        [-1, 0],
        [0, -1]
    ]
    for (let i = 0; i < 内容.length; i++) {
        if (内容[i] == -16777216) {
            x = i % w
            y = parseInt(i / w)
            笔画 = []
            查找(x, y)
            if (笔画.length > 0)
                笔画和.push(笔画)
        }
    }

    function 查找(x, y) {
        let ys = 内容[w * y - w + x]
        if (ys == -16777216) {
            内容[w * y - w + x] = 0
            笔画.push([x, y])
            for (let a = 0; a < 4; a++) {
                查找(x + fx[a][0], y + fx[a][1])
            }
        }
    }
    return 笔画和
}```
