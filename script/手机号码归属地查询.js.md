var numbel=dialogs.rawInput("请输入你要查询的号码：","");

function cutstr(a, b, c, f, e) {
    a = a.split(b);
    var d = ""
    if (e < a.length && e != null) {} else {
        e = a.length;
    }
    if (f == null) {
        f = 1;
    }
    for (i = f; i < e; i++) {
        tmp = a[i].split(c);
        if (tmp.length > 1) {
            d += tmp[0];
        }
    }
    return d;
}```
