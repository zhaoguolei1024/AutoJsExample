function 读取内容(流, 开始, 个数) {
    var buf = java.lang.reflect.Array.newInstance(java.lang.Byte.TYPE, 个数);
    for (let i = 0; i < 个数; i++) {
        buf[i] = 流.get(开始 + i)
    }
    return buf
}

function 写入内容(流, 开始, 内容) {
    var b
    if (typeof(内容) == "string") {
        b = java.lang.String(内容.toString()).getBytes();
    } else {
        b = 内容
    }
    for (let i = 0; i < b.length; i++) {
        流.put(开始 + i, b[i])
    }
    return b.length
