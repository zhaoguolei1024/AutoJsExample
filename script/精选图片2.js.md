function getimg() {
    var url = "http://gank.io/api/data/" + word + "/" + num + "/" + page;
    a = http.get(url).body.json();
    if (a.results[0] == null) {
        c[0] = null;
        return null;
    }
    for (i in a.results) {
        c[i] = a.results[i].url;
    }
    return c;
}
