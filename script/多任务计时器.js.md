function 毫秒格式化(num) {
    var 总秒 = parseInt(num / 1000);
    var 秒 = 总秒 % 60;
    var 总分 = parseInt(总秒 / 60);
    var 分 = 总分 % 60;
    var 总时 = parseInt(总分 / 60);
    var 时 = 总时 % 24;
    var 总天 = parseInt(总时 / 24);
    return pf(总天, 3) + "::" + pf(时, 2) + ":" + pf(分, 2) + ":" + pf(秒, 2);

    function pf(num, length) {
        return (Array(length).join('0') + num).slice(-length);
    }
}
