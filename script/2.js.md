function 点击图片(c) {
    img_base64 = c.posData.imageData.data

function 点击(c) {
    x = Math.round(c.posData.x.split("%")[0] / 100 * 1080)
    y = Math.round(c.posData.y.split("%")[0] / 100 * 2261)
    tim = c.duration
    if (tim = "undefined") {
        tim = 100
    }
    tie = c.delay
    if (tie = "undefined") {
        tie = 1000
    }
    // return [x,y,tim,tie]
    sleep(tie)
    press(x, y, tim)
}

function dq(k) {
    let file = open(k);
    let text = file.read();
    return text;
    file.close();
};
