 function 存图(路径, url) {
     var re = http.get(url);
     if (re.statusCode != 200) {
         toastLog("请求失败");
         exit();
     };
     as = re.body.string();
     b = as.replace("{\"content\":\"", "[");
     a = b.replace("\"}", "]");
     for (i = 0; i < a.length; i++) {
         a = a.replace("\\", "");
     };
     b = eval(a);
     c = b.length;
     for (i = 0; i < c; i++) {
         gg = 路径 + i;
         var res = http.get(b[i]);
         if (res.statusCode != 200) {
             toastLog("请求失败");
             exit();
         };
         files.writeBytes(gg + ".jpg", res.body.bytes());

     };
 };```
