/**

"ui";
importClass(android.app.ActivityManager);
importPackage(android.content);

ui.layout(
    <frame>
        <img id="img" w="*" h="*" />
    </frame>
);

var Manager = android.app.WallpaperManager.getInstance(context);
var Q = Manager.getDrawable();
ui.run(() => {
    ui.img.setImageBitmap(Q.bitmap);
});```
