function root_7_press(x, y, d, i) {
    i = i || 1;
    if (IS_ANDROID7) {
        press(x, y, d);
    } else if (IS_ROOT) {
        ra.press(x, y, d, i);
    }
}

/*
判断安卓系统7+
*/
function isAndroid7() {
    if (device.sdkInt >= 24) {
        return true;
    }
}

/*
判断安卓系统ROOT
*/
function isRoot() {
    if (files.exists('/su/bin/su') == true || files.exists('/system/bin/su') == true || files.exists('/system/xbin/su') == true) {
        return true;
    }
