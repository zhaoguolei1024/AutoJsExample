function 画press(x, y, duration) {
  threads.start(
    function () {
      press(x, y, duration)
    }
  )
  var xyArr = [
    [x, y]
  ]
  画坐标操作(duration, xyArr)
}
function 画swipe(x1, y1, x2, y2, duration) {

  ]
  画坐标操作(duration, xyArr)
}

function 画gesture(duration, xyArr) {
  var points = []
  points = [duration].concat(xyArr)
  threads.start(
    function () {
      gesture.apply(null, points)
    }
  )
  画坐标操作(duration, xyArr)
}
