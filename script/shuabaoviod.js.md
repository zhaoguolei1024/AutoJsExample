/**
 * 随机时间滑动
 */
function randmon_time_swipe(){
    rand_time = random(min , max)*1000;
    //不规则滑行
    x = random(100, device.width-100);//初始点x
    y = random(device.height/2+100,device.height/2+200)//初始点y
    x1= x;
    y1= random(140,200)
    log(x+','+y +';'+x1+','+y1)
    sleep(rand_time)
    swipe(x,y,x1,y1,random(1000,1500));
    count++;
    return count;
}
