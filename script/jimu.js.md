function array_index (arr, str) {
    arr.forEach((ber , index) => {
        if (ber == str || ber.indexOf(str) > -1) {
            return index
        } else {
            return null;
        }
    })
    log('遍历完毕')
    return null;
}

//随机函数
function randomAt (str) {
    if (str.indexOf('|') > -1) {
        v11 = str.split('|')[0]
        v12 = str.split('|')[1]
        if (random(1, 2) == 1) {
            v2 = Number(v11) + random(0, v12)
        } else {
            v2 = Number(v11) - random(0, v12)
        }
        return v2
    } else {
        return Number(str)
    }
}
