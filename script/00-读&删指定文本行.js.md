function read_delete() {
    var path = "/sdcard/1/1.txt"; //txt文本路径
    var  reg  =  /^\s+|\s+$/g; //匹配无效空白行
    var txt = files.read(path).replace(reg, "").split("\n");
    let ret_text = txt[0];
    log(ret_text.length);
    if (txt != "") {
        txt.splice(0, 1);
        files.write(path, txt.join("\n"));
        if (ret_text.length > 0) {
            return ret_text.trim();
        }
    } else {
        return "没有号了";
    }

}```
