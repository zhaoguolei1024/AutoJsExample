10000 + random(1, 15000));



function   关闭程序 () {

    var q1 = 0
    openAppSetting(getPackageName("刷宝短视频"));

    setInterval(function() {


        while (!click("结束运行"));
        q1++;
        while (!click("确定"));

        while (!click("清除数据"));
        while (!click("清除缓存"));
        while (!click("确定"));
        if (q1=>5) {
            exit();
        }

    },
    2000);
