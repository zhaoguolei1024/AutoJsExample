            <img src='{{img_link}}' w='auto' h='auto' id='img_0'></img>

//程序初始化线程
threads.start(function(){
    //请求截屏权限

    }else{
        toastLog('请求截屏权限失败');
    }
});

//截图事件
ui.button_0.on('click', ()=> {
    
});

//运行事件
ui.button_1.on('click', ()=> {
    var config = {};
    config.con = rawInput('请输入','');
    var th = '';
    if(config.con != ''){
        if(th.isAilve()){
            th.interrupt();
        }
        th = threads.start(function(){
            //主函数
            bu_1(config);
        });
    }else{
        toastLog('请输入字符串');
    }
});

function bu_0(obj){
    log('开始bu_1函数,传入数据为:'+obj);
}

    log('开始bu_1函数,传入数据为:'+obj);
}
