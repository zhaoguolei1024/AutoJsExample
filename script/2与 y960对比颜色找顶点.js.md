function zb(clol) {
  var point = findColorInRegion(img, clol, 9, 500, 1070, 1000);
  if (point) {
    dx = point.x;
    dy = point.y;
    toastLog("顶点dx = " + dx + ", dy = " + dy);
  } else {
    toastLog("没有找到");
    exit();
  }
}

function im(x2, y2) {
  //获取在点(x, y)处的颜色
  c = images.pixel(img, x2, y2);
  //显示该颜色
  mr = colors.red(c);
  mg = colors.green(c);
  mb = colors.blue(c);
  return mr, mg, mb, c;
};
