while (1) {
    let data = getDate(name, p);
    p++;
    let d = data.match(/>\S{4}<\/a>/g);
    try {
        for (let i = 0; i < d.length; i++) {
            let n = d[i].split(">")[1].split("<")[0];
            if (dataSource.indexOf(n) == -1) {
                dataSource.push(n);
            }
        }
    } catch (e) {}
    if (d == null) break;
}
